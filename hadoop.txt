hdfs dfs -ls /path List files in directory
hdfs dfs -mkdir /path Create directory
hdfs dfs -mkdir -p /path Create nested directories
hdfs dfs -rmdir /path Remove empty directory
hdfs dfs -rm /file Delete file
hdfs dfs -rm -r /dir Delete directory recursively
hdfs dfs -cp src dst Copy file
hdfs dfs -mv src dst Move file
hdfs dfs -put localfile /hdfs/path Upload file to HDFS
hdfs dfs -get /hdfs/file localpath Download file from HDFS
hdfs dfs -copyFromLocal file /hdfs Copy from local
hdfs dfs -copyToLocal /hdfs/file Copy to local
hdfs dfs -cat /file View file content
hdfs dfs -tail /file View last part of file
hdfs dfs -du /path Disk usage
hdfs dfs -df -h Check HDFS free space
hdfs dfs -count /path Count files, directories
hdfs dfs -chmod 755 /file Change permissions
hdfs dfs -chown user:group /file Change owner
hdfs dfs -setrep 3 /file Set replication factor

yarn node -list List cluster nodes
yarn logs -applicationId <appID> View application logs
